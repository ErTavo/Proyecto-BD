CREATE TABLE CLIENTE (
ID_CLIENTE int IDENTITY PRIMARY KEY NOT NULL,
NOMBRE varchar(255) NOT NULL,
EDAD int NOT NULL,
CORREO varchar(255) NOT NULL);

CREATE TABLE TICKET (
ID_TICKET int IDENTITY PRIMARY KEY NOT NULL,
ID_EVENTO int NOT NULL,
//ID_CLIENTE int NOT NULL);

CREATE TABLE EVENTO (
ID_EVENTO int IDENTITY PRIMARY KEY NOT NULL,
NOMBRE varchar(255) NOT NULL,
FECHA date NOT NULL,
HORA time NOT NULL,
DESCRIPCION varchar(255) NOT NULL);

CREATE TABLE TICKETXCLIENTE (
ID_CLIENTE int NOT NULL,
ID_TICKET int NOT NULL);

ALTER TABLE TICKET ADD FOREIGN KEY (ID_EVENTO) REFERENCES EVENTO (ID_EVENTO);
ALTER TABLE TICKETXCLIENTE ADD FOREIGN KEY (ID_TICKET) REFERENCES TICKET (ID_TICKET);
ALTER TABLE TICKETXCLIENTE ADD FOREIGN KEY (ID_CLIENTE) REFERENCES CLIENTE (ID_CLIENTE);
